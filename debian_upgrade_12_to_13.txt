###
# Upgrade from Debian 12 (bookworm) to 13 (trixie)
###

### Screen
# Use screen -r debian-upgrade to reconnect, if connection broke during the process
# Detach: Ctrl+a, then d
# Close: exit or ctrl+d
screen -S debian-upgrade

### Check
# Check for sufficient free disk space and current version
df -h
df -i
lsb_release -a
uname -a
cat /etc/debian_version

# Check for held packages
apt-mark showhold | more
dpkg --get-selections | grep 'hold$' | more

# Check for read-only mounts
mount | grep ' ro[,)]'

### Installed non-Debian packages
# Identify packages not from Debian repositories
# Note: apt-forktracer must be installed separately: apt install apt-forktracer
apt list '?narrow(?installed, ?not(?origin(Debian)))'
apt-forktracer | sort

### Disable third-party repositories
# To prevent conflicts during the upgrade, comment out all non-Debian repositories
# Check all files in /etc/apt/sources.list.d/ and add a '#' at the beginning of each line
# Example: #deb http://deb.opera.com/opera-beta/ stable non-free

### Update current system + clean
# Update the current system to the latest bookworm packages and clear package cache
apt update
apt upgrade
apt dist-upgrade
apt --purge autoremove
apt clean
find /etc -name '.dpkg-old' -o -name '.ucf-old' -o -name '*.merge-error'

### Backup
mkdir -p ~/debian_upgrade_12_13
# List obsolete packages
apt list '~o' > ~/debian_upgrade_12_13/obsolete_packages_$(date +"%Y%m%d").bkp
# Create a backup of critical system files
tar -cvzf ~/debian_upgrade_12_13/etc_bkp_$(date +"%Y%m%d").tar.gz /etc/
tar -cvzf ~/debian_upgrade_12_13/var_lib_dpkg_bkp_$(date +"%Y%m%d").tar.gz /var/lib/dpkg/
cp /var/lib/apt/extended_states ~/debian_upgrade_12_13/var_lib_apt_extended_states_$(date +"%Y%m%d").bkp
dpkg --get-selections "*" > ~/debian_upgrade_12_13/dpkg_get_selections_$(date +"%Y%m%d").bkp
# Backup aptitude package states if you use aptitude
[ -f /var/lib/aptitude/pkgstates ] && cp /var/lib/aptitude/pkgstates ~/debian_upgrade_12_13/aptitude_pkgstates_$(date +"%Y%m%d").bkp
cp -r /etc/apt ~/debian_upgrade_12_13/etc_apt_$(date +"%Y%m%d").bkp

### APT
# Change the release name from 'bookworm' to 'trixie'
# Important: Ensure that all source lists are valid and free of errors before proceeding.
sed -i".12-13.$(date +"%Y%m%d").bkp" 's/bookworm/trixie/g' /etc/apt/sources.list
# Apply to .list and .sources files separately to avoid errors on non-text files or directories
find /etc/apt/sources.list.d -maxdepth 1 -type f \( -name '*.list' -o -name '*.sources' \) -print0 xargs -0 sed -i'.bak' 's/bookworm/trixie/g'

# Verify the sources look correct before upgrading
apt update
apt-cache policy | head -n 40

### Upgrade
# Important: Pay close attention to configuration file prompts during the upgrade process.
# Use 'diff -u' to review changes and manually merge or keep your version.
# Dry-run first to preview changes
apt -s full-upgrade
apt upgrade --without-new-pkgs
apt full-upgrade

### Sanity check
dpkg --audit
dpkg --configure -a
apt-get -f install
# Check for partially installed or residual packages
dpkg -l | awk '$1 ~ /^(iF|iU|iH|rc)$/ {print}'

### Pre-reboot check
# Important: Before rebooting, ensure critical packages for remote access are installed and functional.
# A lost SSH connection after reboot without these packages means losing remote access entirely.
apt install --yes openssh-server sudo screen
systemctl is-active ssh || systemctl start ssh
systemctl is-enabled ssh || systemctl enable ssh

### Reboot
reboot

### Check again
# Verify the upgrade was successful
lsb_release -a
uname -a
cat /etc/debian_version

# Check for failed systemd units
systemctl --failed

# Check running services (adjust to your stack)
systemctl status ssh
# systemctl status nginx
# systemctl status apache2
# systemctl status postgresql
# systemctl status mariadb
# systemctl status docker
# systemctl status fail2ban

# Check listening ports
ss -tulpn

# Check for processes still using old library versions (server: restart services instead of full reboot where possible)
# Install if needed: apt install needrestart
needrestart

# Check for any post-upgrade boot warnings
journalctl -b -p warning | tail -n 200

# Check logs for apt/dpkg errors
grep -iE 'error|failed|dependency' /var/log/apt/term.log | tail -n 50

# Check disk and inode usage after upgrade
df -h
df -i

# Verify file integrity of installed packages (install if needed: apt install debsums)
# Note: /etc differences are expected (intentional config changes); focus on /usr/bin, /usr/lib
debsums -s

# Update GRUB if kernel was upgraded
update-grub

### Final update
# Perform a final cleanup and update
apt update
apt full-upgrade
apt --purge autoremove
apt autoclean

# Convert source files to the new 'deb822' format
apt modernize-sources

# If you encounter dependency issues, run this command
# sudo apt --fix-broken install

### Re-enable third-party repositories
# After the core upgrade is complete, re-enable your third-party repositories by removing the '#' from the commented lines in /etc/apt/sources.list.d/
# Then, update and upgrade your system again to pull in the latest versions of these packages
apt update
apt upgrade
